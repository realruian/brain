```go
-- 第一步：找出使用次数最多的前50个背景URL
WITH top_backgrounds AS (
    SELECT 
        get_json_object(decorate_config, '$.background.previewUrl') AS background_url,
        COUNT(1) AS usage_count
    FROM origindb_ss.instorelive_ai_live__live_show
    WHERE dt = "20250806"
    AND start_time > "2025-04-01 00:00:00"
    GROUP BY get_json_object(decorate_config, '$.background.previewUrl')
    ORDER BY usage_count DESC
    LIMIT 50
```
